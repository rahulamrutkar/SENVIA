🔐 Security
📊 Google Sheets — Automatic Backup

Every product you add and every customer inquiry is automatically sent to your Google Sheet as a permanent record.

How it works:
• Products & inquiries are saved in this app (localStorage) first
• They are also silently posted to your Sheet in the background
• Open your Sheet anytime at sheets.google.com to see all data

ℹ️ Note: Due to browser security rules, the app cannot read back from Sheets — it only writes to it. This is normal and expected.

Paste the URL after deploying the script below
📱 Make changes visible on ALL devices
After connecting Sheets, click below to download an updated index.html with your Sheets URL baked in. Upload this file to GitHub — from that moment, every device in the world will sync your latest products, categories, and settings automatically.
After download: replace your file on GitHub with this one → Netlify auto-redeploys → done.
Step-by-step setup:
  1. Open Google Sheets → create a new spreadsheet → name it SENVIA
  2. In the menu: Extensions → Apps Script
  3. Delete all existing code → paste the entire script below
  4. Click Deploy → New deployment
  5. Type: Web App | Execute as: Me | Who has access: Anyone
  6. Click Deploy → Authorise access (very important!) → copy the Web App URL
  7. Paste that URL above → Save URL → click Test Connection
  8. Click ⬇️ Download Updated App → upload to GitHub → done!
⚠️ Most common reason for errors: After clicking Deploy, Google shows an "Authorisation required" screen — you must click Review permissions → Allow. If you skipped this, re-deploy and authorise.
// ════════════════════════════════════════════════ // SENVIA — Google Apps Script (REPLACE ENTIRE FILE WITH THIS) // Deploy → New deployment → Web App // Execute as: Me | Who has access: Anyone // ════════════════════════════════════════════════ // ── READ: JSONP endpoint — called by every browser on page load ── // Returns ALL shared data: products, categories, settings, about, testimonials function doGet(e) { var cb = e.parameter.callback || 'callback'; var action = e.parameter.action || ''; if (action === 'getAllData') { // Each getter wrapped independently — one failure never blanks the whole app var data = { products: [], cats: [], cfg: {}, about: {}, testimonials: [] }; try { data.products = getProductRows(); } catch(ex) { data._errProducts = ex.toString(); } try { data.cats = getCategoryRows(); } catch(ex) { data._errCats = ex.toString(); } try { data.cfg = getSettingsRow(); } catch(ex) { data._errCfg = ex.toString(); } try { data.about = getAboutRow(); } catch(ex) { data._errAbout = ex.toString(); } try { data.testimonials = getTestimonialRows(); } catch(ex) { data._errTest = ex.toString(); } var json = JSON.stringify(data); return ContentService .createTextOutput(cb + '(' + json + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } if (action === 'getAbout') { // Read all Settings rows and return only the about_* keys as a clean object var about = {}; try { var cfg = getSettingsRow(); Object.keys(cfg).forEach(function(k){ if(k.indexOf('about_') === 0){ about[k.replace('about_', '')] = cfg[k]; } }); } catch(ex) {} var json = JSON.stringify(about); return ContentService .createTextOutput(cb + '(' + json + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } if (action === 'getInquiries') { var rows = []; try { rows = getInquiryRows(); } catch(ex) {} var json = JSON.stringify(rows); return ContentService .createTextOutput(cb + '(' + json + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService .createTextOutput('SENVIA API OK') .setMimeType(ContentService.MimeType.TEXT); } // ── WRITE: POST endpoint — called silently in background ── function doPost(e) { try { var raw = e.postData.contents; var d = JSON.parse(raw); var act = d.action || ''; if (act === 'saveProduct') return saveProduct(d.data); if (act === 'saveAllProducts') return saveAllProducts(d.data); if (act === 'deleteProduct') return deleteProduct(d.data); if (act === 'saveCategories') return saveCategories(d.data); if (act === 'saveSettings') return saveSettings(d.data); if (act === 'addInquiry') return addInquiry(d.data); if (act === 'saveAbout') return saveAbout(d.data); if (act === 'saveAllTestimonials') return saveAllTestimonials(d.data); if (act === 'ping') return ok('pong'); return ok('unknown: ' + act); } catch(err) { return ok('error: ' + err.toString()); } } function ok(msg) { return ContentService .createTextOutput(JSON.stringify({status:'ok',msg:msg})) .setMimeType(ContentService.MimeType.JSON); } // ── Sheet helpers ────────────────────────────── function getSheet(name, headers) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sh = ss.getSheetByName(name); if (!sh) { sh = ss.insertSheet(name); sh.getRange(1, 1, 1, headers.length) .setValues([headers]) .setFontWeight('bold') .setBackground('#1c1c1c') .setFontColor('#d4a853'); sh.setFrozenRows(1); } return sh; } // ── READ helpers ─────────────────────────────── function getProductRows() { var sh = getSheet('Products', ['id','name','category','price','description','image','createdAt']); var vals = sh.getDataRange().getValues(); if (vals.length <= 1) return []; var hdrs = vals[0]; return vals.slice(1) .filter(function(r){ return String(r[0]).trim() !== ''; }) .map(function(r){ var o = {}; hdrs.forEach(function(h,i){ o[h] = String(r[i]||''); }); return o; }); } function getCategoryRows() { var sh = getSheet('Categories', ['category']); var vals = sh.getDataRange().getValues(); if (vals.length <= 1) return []; return vals.slice(1) .map(function(r){ return String(r[0]||'').trim(); }) .filter(function(c){ return c !== ''; }); } function getSettingsRow() { var sh = getSheet('Settings', ['key','value']); var vals = sh.getDataRange().getValues(); if (vals.length <= 1) return {}; var cfg = {}; vals.slice(1).forEach(function(r){ if(r[0]) cfg[String(r[0])] = String(r[1]||''); }); return cfg; } function getInquiryRows() { var sh = getSheet('Inquiries', ['Timestamp','Name','Email','Phone','Product','Message','ID']); var vals = sh.getDataRange().getValues(); if (vals.length <= 1) return []; var hdrs = vals[0]; return vals.slice(1) .filter(function(r){ return String(r[1]||'').trim() !== ''; }) .map(function(r){ var o = {}; hdrs.forEach(function(h,i){ o[String(h)] = String(r[i]||''); }); return { id: o['ID'] || '', name: o['Name'] || '', email: o['Email'] || '', phone: o['Phone'] || '', productName: o['Product'] || 'General', message: o['Message'] || '', timestamp: o['Timestamp'] || '' }; }) .reverse(); } function getAboutRow() { var sh = getSheet('About', ['key','value']); var vals = sh.getDataRange().getValues(); if (vals.length <= 1) return {}; var about = {}; vals.slice(1).forEach(function(r){ if(r[0]) about[String(r[0])] = String(r[1]||''); }); return about; } function getTestimonialRows() { var sh = getSheet('Testimonials', ['id','name','role','rating','review','image']); var vals = sh.getDataRange().getValues(); if (vals.length <= 1) return []; var hdrs = vals[0]; return vals.slice(1) .filter(function(r){ return String(r[0]).trim() !== ''; }) .map(function(r){ var o = {}; hdrs.forEach(function(h,i){ o[h] = String(r[i]||''); }); return o; }); } // ── WRITE helpers ────────────────────────────── function saveProduct(d) { var sh = getSheet('Products', ['id','name','category','price','description','image','createdAt']); var vals = sh.getDataRange().getValues(); for (var i = 1; i < vals.length; i++) { if (String(vals[i][0]) === String(d.id)) { sh.getRange(i+1, 1, 1, 7).setValues([[ d.id, d.name, d.category, d.price||'', d.description||'', d.image||'', d.createdAt||'' ]]); return ok('updated'); } } sh.appendRow([d.id, d.name, d.category, d.price||'', d.description||'', d.image||'', d.createdAt||'']); return ok('added'); } function saveAllProducts(d) { var sh = getSheet('Products', ['id','name','category','price','description','image','createdAt']); if (sh.getLastRow() > 1) { sh.getRange(2, 1, sh.getLastRow()-1, 7).clearContent(); } if (d.products && d.products.length > 0) { var rows = d.products.map(function(p){ return [p.id||'', p.name||'', p.category||'', p.price||'', p.description||'', p.image||'', p.createdAt||'']; }); sh.getRange(2, 1, rows.length, 7).setValues(rows); } return ok('all products saved'); } function deleteProduct(d) { var sh = getSheet('Products', ['id','name','category','price','description','image','createdAt']); var vals = sh.getDataRange().getValues(); for (var i = 1; i < vals.length; i++) { if (String(vals[i][0]) === String(d.id)) { sh.deleteRow(i + 1); return ok('deleted'); } } return ok('not found'); } function saveCategories(d) { var sh = getSheet('Categories', ['category']); if (sh.getLastRow() > 1) { sh.getRange(2, 1, sh.getLastRow()-1, 1).clearContent(); } if (d.cats && d.cats.length > 0) { var rows = d.cats.map(function(c){ return [c]; }); sh.getRange(2, 1, rows.length, 1).setValues(rows); } return ok('categories saved'); } function saveSettings(d) { var sh = getSheet('Settings', ['key','value']); // Accept any key dynamically (storeName, tagline, logoUrl, about_*, etc.) var keys = Object.keys(d); keys.forEach(function(k){ if (k === 'apiUrl') return; // never store the API URL in Sheets var val = d[k]; if (val === undefined || val === null) return; var vals = sh.getDataRange().getValues(); for (var i = 1; i < vals.length; i++) { if (String(vals[i][0]) === k) { sh.getRange(i+1, 2).setValue(String(val)); return; } } sh.appendRow([k, String(val)]); }); return ok('settings saved'); } function addInquiry(d) { var sh = getSheet('Inquiries', ['Timestamp','Name','Email','Phone','Product','Message','ID']); sh.appendRow([ new Date(), d.name||'', d.email||'', d.phone||'', d.productName||'General', d.message||'', d.id||'' ]); return ok('inquiry saved'); } function saveAbout(d) { var sh = getSheet('About', ['key','value']); var keys = ['title','subtitle','content','address','phone','email']; keys.forEach(function(k){ if (d[k] === undefined) return; var vals = sh.getDataRange().getValues(); for (var i = 1; i < vals.length; i++) { if (String(vals[i][0]) === k) { sh.getRange(i+1, 2).setValue(d[k]); return; } } sh.appendRow([k, d[k]]); }); return ok('about saved'); } function saveAllTestimonials(d) { var sh = getSheet('Testimonials', ['id','name','role','rating','review','image']); if (sh.getLastRow() > 1) { sh.getRange(2, 1, sh.getLastRow()-1, 6).clearContent(); } if (d.testimonials && d.testimonials.length > 0) { var rows = d.testimonials.map(function(t){ // Truncate image to 40000 chars to stay within Sheets cell limit var img = String(t.image||''); if(img.length > 40000) img = img.substring(0, 40000); return [t.id||'', t.name||'', t.role||'', t.rating||'5', t.review||'', img]; }); sh.getRange(2, 1, rows.length, 6).setValues(rows); } return ok('testimonials saved'); }
